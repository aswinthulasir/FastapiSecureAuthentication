CREATE TABLE hotel_type (
    u_type INT PRIMARY KEY,
    type_name VARCHAR(50) NOT NULL
);

CREATE TABLE dish_types (
    type_id INT PRIMARY KEY,
    type_name VARCHAR(50) NOT NULL
);

CREATE TABLE food_types (
    food_id INT PRIMARY KEY,
    food_name VARCHAR(50) NOT NULL
);

INSERT INTO food_types VALUES

CREATE TABLE hotel_items (
    item_id SERIAL PRIMARY KEY,
    u_id VARCHAR(16) NOT NULL,
    dish_name VARCHAR(100) NOT NULL,
    dish_type INT NOT NULL,
    dish_description TEXT,
    img_1 TEXT,
    img_2 TEXT,
    rate NUMERIC(10, 2) NOT NULL,
    food_type INT NOT NULL,
    status BOOLEAN DEFAULT TRUE, -- availability status
    FOREIGN KEY (u_id) REFERENCES users(u_id),
    FOREIGN KEY (dish_type) REFERENCES dish_types(type_id),
    FOREIGN KEY (food_type) REFERENCES food_types(food_id)
);
